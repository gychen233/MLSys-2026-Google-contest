int div_ceil(int n, int m) {
    return (n + m - 1) / m;
}

struct Tensor {
    int height;
    int width;
    int input;
    std::vector<int> outputs;
    bool is_cut;
    bool is_input;
    Tensor(): height(0), width(0), input(0), is_cut(0), is_input(0) {}
    Tensor(int height, int width) {
        this->height = height;
        this->width = width;
    }
    Tensor(int height, int width, int input) {
        this->height = height;
        this->width = width;
        this->input = input;
        this->is_cut = 1;
        this->is_input = 0;
    }
};

struct Node {
    float cost;
    bool is_matmul;
    std::vector<int> inputs;
    std::vector<int> outputs;
};

class ScheduleEvaluator {
private:
    int n, m;
    std::vector<Tensor> tensors;
    std::vector<Node> nodes;
    int fast_memory_capacity;
    float slow_memory_bandwidth;
    std::vector<int> native_granularity;
    std::vector<std::vector<int> > d_in, d_out;
    std::vector<int> total_d_in, total_d_out;
    std::vector<int> topo_id;
    int n_layers;
    std::vector<std::vector<int> > subgraphs;
    std::vector<std::vector<int> > granularities;
    std::vector<std::vector<int> > tensors_to_retain;
    std::vector<std::vector<int> > traversal_orders;
    std::vector<bool> retain;
    std::vector<bool> granularity_orders;
    std::vector<float> lantencies;
public:
    ScheduleEvaluator() : n(0), m(0), fast_memory_capacity(0), slow_memory_bandwidth(0.) {}

    ScheduleEvaluator (
        const std::vector<int>& widths,
        const std::vector<int>& heights,
        const std::vector<std::vector<int> >& inputs,
        const std::vector<std::vector<int> >& outputs,
        const std::vector<float>& base_costs,
        const std::vector<std::string>& op_types,
        const int fast_memory_capacity,
        const float slow_memory_bandwidth,
        const std::vector<int>& native_granularity
    ) {
        this->fast_memory_capacity = fast_memory_capacity;
        this->slow_memory_bandwidth = slow_memory_bandwidth;
        this->native_granularity = native_granularity;

        this->n = widths.size();
        assert(this->n == heights.size());
        for (int i = 0; i < this->n; i++) {
            Tensor t;
            t.width = widths[i];
            t.height = heights[i];
            t.is_input = 0;
            this->tensors.push_back(t);
        }

        this->m = inputs.size();
        assert(this->m == outputs.size());
        assert(this->m == base_costs.size());
        assert(this->m == op_types.size());
        for (int i = 0; i < m; i++) {
            Node o;
            o.inputs = inputs[i];
            o.outputs = outputs[i];
            o.cost = base_costs[i];
            if (op_types[i] == "MatMul") o.is_matmul = 1;
            else o.is_matmul = 0;
            nodes.push_back(o);
            for (int j = 0; j < o.inputs.size(); j++) {
                tensors[o.inputs[j]].outputs.push_back(i);
            }
            for (int j = 0; j < o.outputs.size(); j++) {
                tensors[o.outputs[j]].input = i;
                tensors[o.outputs[j]].is_input = 0;
            }
        }

        topo_id.resize(m);
        int d[MAX_N];
        init(d);
        for (int i = 0; i < n; i++) {
            Tensor t = tensors[i];
            for (int j = 0; j < t.outputs.size(); j++) {
                d[t.outputs[j]]++;
            }
        }
        std::queue<int> q;
        for (int i = 0; i < m; i++) {
            if (d[i] == 0) {
                q.push(i);
            }
        }
        int cnt = 0;
        while(q.size()) {
            int x = q.front();
            q.pop();
            topo_id[x] = ++cnt;
            Node nd = nodes[x];
            for (int i = 0; i < nodes[x].outputs.size(); i++) {
                Tensor t = tensors[nodes[x].outputs[i]];
                for (int j = 0; j < t.outputs.size(); j++) {
                    d[t.outputs[j]]--;
                    if (d[t.outputs[j]] == 0) {
                        q.push(t.outputs[j]);
                    }
                }
            }
        }
    }

    void get_necessity() {
        n_layers = subgraphs.size();
        d_in.resize(n_layers);
        d_out.resize(n_layers);
        total_d_in.assign(n, 0);
        total_d_out.assign(n, 0);
        for (int i = 0; i < subgraphs.size(); i++) {
            d_in[i].assign(n, 0);
            d_out[i].assign(n, 0);
            for (int j = 0; j < subgraphs[i].size(); j++) {
                Node nd = nodes[subgraphs[i][j]];
                for (int k = 0; k < nd.outputs.size(); k++) {
                    d_in[i][nd.outputs[0]]++;
                    total_d_in[nd.outputs[0]]++;
                }
                for (int k = 0; k < nd.inputs.size(); k++) {
                    d_out[i][nd.inputs[k]]++;
                    total_d_out[nd.inputs[k]]++;
                }
            }
        }
    }

    int fd_fa(int x, int* fa) {
        if (x != fa[x]) {
            fa[x] = fd_fa(fa[x], fa);
        }
        return fa[x];
    }

    bool get_subgraphs() {
        int fa[MAX_M];
        for (int i = 0; i < m; i++) {
            fa[i] = i;
        }
        for (int i = 0; i < n; i++) {
            Tensor t = tensors[i];
            if (t.is_input) {
                continue;
            }
            if (!t.is_cut) {
                int x = fd_fa(t.input, fa);
                for (int j = 0; j < t.outputs.size(); j++) {
                    fa[fd_fa(t.outputs[j], fa)] = x;
                }
            }
        }
        int d[MAX_M];
        init(d);
        std::vector<std::vector<int> > out;
        out.resize(m);
        for (int i = 0; i < n; i++) {
            Tensor t = tensors[i];
            if (t.is_input) {
                continue;
            }
            if (t.is_cut) {
                int x = fd_fa(t.input, fa);
                for (int j = 0; j < t.outputs.size(); j++) {
                    int y = fd_fa(t.outputs[j], fa);
                    if (x != y) {
                        out[x].push_back(y);
                        d[y]++;
                    }
                }
            }
        }
        int order[MAX_M], cnt = 0;
        std::deque<int> dq;
        for (int i = 0; i < m; i++) {
            if (fa[i] == i) {
                if (d[i] == 0) {
                    dq.push_front(i);
                }
            }
        }
        while(dq.size()) {
            int x = dq.front();
            dq.pop_front();
            order[cnt++] = x;
            bool flag = 0;
            for (int i = 0; i < out[x].size(); i++) {
                d[out[x][i]]--;
                if (d[out[x][i]] == 0) {
                    if (!flag) {
                        dq.push_front(out[x][i]);
                        flag = 1;
                    } else {
                        dq.push_back(out[x][i]);
                    }
                }
            }
        }

        // bottle neck
        subgraphs.clear();
        for (int i = 0; i < cnt; i++) {
            std::vector<int> subgraph;
            for (int j = 0; j < m; j++) {
                if (fd_fa(j, fa) == order[i]) {
                    subgraph.push_back(j);
                }
            }
            // 由于有额外的 this 键字，所以要写成 lambda 表达式
            sort(subgraph.begin(), subgraph.end(), [this](int x, int y) {
                return this->topo_id[x] < this->topo_id[y];
            });
            subgraphs.push_back(subgraph);
        }

        // for (int i = 0; i < subgraphs.size(); i++) {
        //     for (int j = 0; j < subgraphs[i].size(); j++) {
        //         printf("%d ", subgraphs[i][j]);
        //     }
        //     printf("\n");
        // }

        get_necessity();

        // Improve the inner order of subgraphs
        std::vector<int> temp_subgraphs;
        for (int i = 0; i < subgraphs.size(); i++) {
            temp_subgraphs.clear();
            for (int j = 0; j < subgraphs[i].size(); j++) {
                if (d_out[i][nodes[subgraphs[i][j]].outputs[0]] != 0) {
                    temp_subgraphs.push_back(subgraphs[i][j]);
                }
            }
            if (temp_subgraphs.size() != subgraphs[i].size() - 1) {
                // printf("Multiple outputs!\n");
                return 0;
            }
            for (int j = 0; j < subgraphs[i].size(); j++) {
                if (d_out[i][nodes[subgraphs[i][j]].outputs[0]] == 0) {
                    temp_subgraphs.push_back(subgraphs[i][j]);
                }
            }
            subgraphs[i] = temp_subgraphs;

        }

        // Check the legitimacy
        bool used_op[MAX_M];
        bool is_computed[MAX_N];
        init(used_op);
        init(is_computed);
        for (int i = 0; i < subgraphs.size(); i++) {
            for (int j = 0; j < subgraphs[i].size(); j++) {
                used_op[subgraphs[i][j]] = 1;
            }
        }
        for (int i = 0; i < m; i++) {
            if (!used_op[i]) {
                // printf("!used_op[i]\n");
                return 0; // assert
            }
        }
        for (int i = 0; i < n; i++) {
            is_computed[i] = 1;
        }
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < this->nodes[i].outputs.size(); j++) {
                is_computed[nodes[i].outputs[j]] = 0;
            }
        }

        bool ephemeral_computed[MAX_N];
        for (int i = 0; i < subgraphs.size(); i++) {
            init(ephemeral_computed);
            for (int j = 0; j < subgraphs[i].size(); j++) {
                Node nd = nodes[subgraphs[i][j]];
                for (int k = 0; k < nd.inputs.size(); k++) {
                    if (d_in[i][nd.inputs[k]] > 0) {
                        if (!ephemeral_computed[nd.inputs[k]]) {
                            // printf("!ephemeral_computed[nd.inputs[k]]\n");
                            return 0; // assert
                        }
                    }
                    else {
                        if (!is_computed[nd.inputs[k]]) {
                            // printf("!is_computed[nd.inputs[k]]\n");
                            return 0; // assert
                        }
                    }
                }
                for (int k = 0; k < nd.outputs.size(); k++) {
                    ephemeral_computed[nd.outputs[k]] = 1;
                }
            }
            for (int j = 0; j < subgraphs[i].size(); j++) {
                Node nd = nodes[subgraphs[i][j]];
                for (int k = 0; k < nd.outputs.size(); k++) {
                    if (d_out[i][nodes[subgraphs[i][j]].outputs[k]] == 0) {
                        is_computed[nodes[subgraphs[i][j]].outputs[k]] = 1;
                    }
                }
            }
        }
        return 1;
    }

    void upd_retain(int layer) {
        tensors_to_retain[layer].clear();
        if (retain[layer]) {
            tensors_to_retain[layer].push_back(nodes[subgraphs[layer].back()].outputs[0]);
        }
    }

    void get_retain() {
        tensors_to_retain.resize(n_layers);
        for (int layer = 0; layer < n_layers; layer++) {
            upd_retain(layer);
        }
    }

    int get_point(int n, int m, int x, int y) {
        return x * m + y;
    }

    void upd_traversal_order(int layer) {
        int n_h = tensors[nodes[subgraphs[layer].back()].outputs[0]].height / granularities[layer][0];
        int n_w = tensors[nodes[subgraphs[layer].back()].outputs[0]].width / granularities[layer][1];

        traversal_orders[layer].clear();

        if (n_h > 1 && n_w > 1) {
            float h_lantency, w_lantency;
            traversal_orders[layer].clear();
            for (int i = 0; i < n_h / 2; i++) {
                for (int j = 0; j < n_w; j++) {
                    traversal_orders[layer].push_back(get_point(n_h, n_w, 2 * i, j));
                }
                for (int j = n_w - 1; j >= 0; j--) {
                    traversal_orders[layer].push_back(get_point(n_h, n_w, 2 * i + 1, j));
                }
            }
            h_lantency = get_lantency(layer);
            
            traversal_orders[layer].clear();
            for (int j = 0; j < n_w / 2; j++) {
                for (int i = 0; i < n_h; i++) {
                    traversal_orders[layer].push_back(get_point(n_h, n_w, i, 2 * j));
                }
                for (int i = n_h - 1; i >= 0; i--) {
                    traversal_orders[layer].push_back(get_point(n_h, n_w, i, 2 * j + 1));
                }
            }
            w_lantency = get_lantency(layer);

            if (h_lantency < w_lantency) {
                traversal_orders[layer].clear();
                for (int i = 0; i < n_h / 2; i++) {
                    for (int j = 0; j < n_w; j++) {
                        traversal_orders[layer].push_back(get_point(n_h, n_w, 2 * i, j));
                    }
                    for (int j = n_w - 1; j >= 0; j--) {
                        traversal_orders[layer].push_back(get_point(n_h, n_w, 2 * i + 1, j));
                    }
                }
            }
        }
    }

    void get_traversal_orders() {
        traversal_orders.resize(n_layers);
        for (int layer = 0; layer < n_layers; layer++) {
            upd_traversal_order(layer);
        }
    }

    int cal_memory(std::vector<int> granularity, int layer) {
        int h[MAX_N][2], w[MAX_N][2];
        bool is_h[MAX_N][2];
        init(h);init(w);
        init(is_h);
        int n_h = tensors[nodes[subgraphs[layer].back()].outputs[0]].height / granularity[0];
        int n_w = tensors[nodes[subgraphs[layer].back()].outputs[0]].width / granularity[1];
        int n_k = nodes[subgraphs[layer].back()].is_matmul ? tensors[nodes[subgraphs[layer].back()].inputs[0]].width / granularity[2] : 1;
        for (int i = subgraphs[layer].size() - 1; i >= 0; i--) {
            Node nd = nodes[subgraphs[layer][i]];
            if (i == subgraphs[layer].size() - 1) {
                if (granularity[0] > native_granularity[0] || granularity[1] > native_granularity[1]) {
                    // printf("Exceed native_granularity!\n");
                    return INF_LANTENCY;
                }
                if (nd.is_matmul) {
                    h[nd.inputs[0]][0] = h[nd.inputs[0]][1] = granularity[0];
                    w[nd.inputs[0]][0] = w[nd.inputs[0]][1] = granularity[2];
                    is_h[nd.inputs[0]][0] = is_h[nd.inputs[0]][1] = 1;
                    h[nd.inputs[1]][0] = h[nd.inputs[1]][1] = granularity[2];
                    w[nd.inputs[1]][0] = w[nd.inputs[1]][1] = granularity[1];
                    is_h[nd.inputs[1]][0] = is_h[nd.inputs[1]][1] = 0;
                } else {
                    for (int j = 0; j < nd.inputs.size(); j++) {
                        h[nd.inputs[j]][0] = h[nd.inputs[j]][1] = granularity[0];
                        w[nd.inputs[j]][0] = w[nd.inputs[j]][1] = granularity[1];
                    }
                }
            } else {
                if (h[nd.outputs[0]][0] > native_granularity[0] || w[nd.outputs[0]][1] > native_granularity[1]) {
                    // printf("Exceed native_granularity!\n");
                    return INF_LANTENCY;
                }
                if (nd.is_matmul) {
                    for (int j = 0; j <= 1; j++) {
                        if (h[nd.outputs[0]][j] > h[nd.inputs[0]][0]) {
                            h[nd.inputs[0]][0] = h[nd.outputs[0]][j];
                            w[nd.inputs[0]][0] = tensors[nd.inputs[0]].width;
                            is_h[nd.inputs[0]][0] = is_h[nd.outputs[0]][j];
                        }
                        h[nd.inputs[0]][1] = h[nd.outputs[0]][j];
                        w[nd.inputs[0]][1] = tensors[nd.inputs[0]].width;
                        is_h[nd.inputs[0]][1] = is_h[nd.outputs[0]][j];


                        h[nd.inputs[1]][0] = tensors[nd.inputs[1]].height;
                        w[nd.inputs[1]][0] = w[nd.outputs[0]][j];
                        is_h[nd.inputs[1]][0] = is_h[nd.outputs[0]][j];
                        if (w[nd.outputs[0]][j] > w[nd.inputs[1]][1]) {
                            h[nd.inputs[1]][1] = tensors[nd.inputs[1]].height;
                            w[nd.inputs[1]][1] = w[nd.outputs[0]][j];
                            is_h[nd.inputs[1]][1] = is_h[nd.outputs[0]][j];
                        }
                    }
                } else {
                    for (int j = 0; j <= 1; j++) {
                        for (int k = 0; k < nd.inputs.size(); k++) {
                            if (h[nd.outputs[0]][j] > h[nd.inputs[k]][0]) {
                                h[nd.inputs[k]][0] = h[nd.outputs[0]][j];
                                w[nd.inputs[k]][0] = w[nd.outputs[0]][j];
                                is_h[nd.inputs[k]][0] = is_h[nd.outputs[0]][j];
                            }

                            if (w[nd.outputs[0]][j] > w[nd.inputs[k]][1]) {
                                h[nd.inputs[k]][1] = h[nd.outputs[0]][j];
                                w[nd.inputs[k]][1] = w[nd.outputs[0]][j];
                                is_h[nd.inputs[k]][1] = is_h[nd.outputs[0]][j];
                            }
                        }
                    }
                }
            }
        }

        bool retained[MAX_N];
        init(retained);
        int retain_memory = 0;
        if (layer > 0) {
            for (int i = 0; i < tensors_to_retain[layer - 1].size(); i++) {
                retained[tensors_to_retain[layer - 1][i]] = 1;
                retain_memory += tensors[tensors_to_retain[layer - 1][i]].height * tensors[tensors_to_retain[layer - 1][i]].width;
            }
        }
        for (int i = 0; i < tensors_to_retain[layer].size(); i++) {
            retained[tensors_to_retain[layer][i]] = 1;
            retain_memory += tensors[tensors_to_retain[layer][i]].height * tensors[tensors_to_retain[layer][i]].width;
        }

        int global_memory = 0;
        for (int i = 0; i < n; i++) {
            if (d_in[layer][i] == 0 && d_out[layer][i] > 0 && !retained[i]) {
                if (h[i][0] > h[i][1] && w[i][0] < w[i][1]) {
                    global_memory += h[i][0] * w[i][0] + h[i][1] * w[i][1];
                } else {
                    global_memory += std::max(h[i][0], h[i][1]) * std::max(w[i][0], w[i][1]);
                }
            }
        }

        int output_memory = 0;
        int conflict_memory = 0;
        Node nd = nodes[subgraphs[layer].back()];
        if (nd.is_matmul && tensors[nd.inputs[0]].width > granularity[2] && !retained[nd.outputs[0]]) {
            output_memory += granularity[0] * granularity[1];
        }

        else {
            if (nd.is_matmul) {
                conflict_memory = std::max(conflict_memory, granularity[0] * granularity[1]
                                + (granularity[0] + granularity[1]) * granularity[2]);
            } else {
                conflict_memory = std::max(conflict_memory, granularity[0] * granularity[1] * ((int)nd.inputs.size() + 1));
            }
        }
        return std::max(retain_memory + output_memory + global_memory, conflict_memory);
    }

    std::vector<int> get_granularity(int layer) {
        std::vector<int> granularity;

        int k_lim = 1;
        Node nd = nodes[subgraphs[layer].back()];
        if (nd.is_matmul) {
            k_lim = std::max(k_lim, tensors[nd.inputs[0]].width);
        }

        int H, W;
        H = native_granularity[0];
        W = native_granularity[1];
        granularity = {H, W, 1};
        if (cal_memory(granularity, layer) <= fast_memory_capacity) {
            for (int K = k_lim; K >= 1; K >>= 1) {
                granularity[2] = K;
                if (cal_memory(granularity, layer) <= fast_memory_capacity) {
                    return granularity;
                }
            }
        }
        if (granularity_orders[layer]) {
            for (int H = native_granularity[0]; H >= native_granularity[0] / 2; H >>= 1) {
                for (int W = native_granularity[1]; W >= native_granularity[1] / 2; W >>= 1) {
                    granularity = {H, W, 1};
                    if (cal_memory(granularity, layer) <= fast_memory_capacity) {
                        for (int K = k_lim; K >= 1; K >>= 1) {
                            granularity = {H, W, K};
                            if (cal_memory(granularity, layer) <= fast_memory_capacity) {
                                return granularity;
                            }
                        }
                    }
                }
            }
        } else {
            for (int W = native_granularity[1]; W >= native_granularity[1] / 2; W >>= 1) {
                for (int H = native_granularity[0]; H >= native_granularity[0] / 2; H >>= 1) {
                    granularity = {H, W, 1};
                    if (cal_memory(granularity, layer) <= fast_memory_capacity) {
                        for (int K = k_lim; K >= 1; K >>= 1) {
                            granularity = {H, W, K};
                            if (cal_memory(granularity, layer) <= fast_memory_capacity) {
                                return granularity;
                            }
                        }
                    }
                }
            }
        }

        return {};
    }

    bool get_granularities() {
        granularities.resize(n_layers);
        for (int layer = 0; layer < n_layers; layer++) {
            granularities[layer].clear();
            granularities[layer] = get_granularity(layer);
            if (granularities[layer].size() == 0) {
                return 0;
            }
        }
        return 1;
    }

    float get_lantency(int layer) {
        int h[MAX_N][2], w[MAX_N][2];
        bool is_h[MAX_N][2];
        bool retained[MAX_N];
        float lantency = 0.;
        float cal_lantency = 0.;
        init(h);init(w);
        init(is_h);
        int n_h = tensors[nodes[subgraphs[layer].back()].outputs[0]].height / granularities[layer][0];
        int n_w = tensors[nodes[subgraphs[layer].back()].outputs[0]].width / granularities[layer][1];
        int n_k = nodes[subgraphs[layer].back()].is_matmul ? tensors[nodes[subgraphs[layer].back()].inputs[0]].width / granularities[layer][2] : 1;
        for (int i = subgraphs[layer].size() - 1; i >= 0; i--) {
            Node nd = nodes[subgraphs[layer][i]];
            if (i == subgraphs[layer].size() - 1) {
                if (nd.is_matmul) {
                    h[nd.inputs[0]][0] = h[nd.inputs[0]][1] = granularities[layer][0];
                    w[nd.inputs[0]][0] = w[nd.inputs[0]][1] = granularities[layer][2];
                    is_h[nd.inputs[0]][0] = is_h[nd.inputs[0]][1] = 1;
                    h[nd.inputs[1]][0] = h[nd.inputs[1]][1] = granularities[layer][2];
                    w[nd.inputs[1]][0] = w[nd.inputs[1]][1] = granularities[layer][1];
                    is_h[nd.inputs[1]][0] = is_h[nd.inputs[1]][1] = 0;
                    cal_lantency += (float)nd.cost / n_k;
                } else {
                    for (int j = 0; j < nd.inputs.size(); j++) {
                        h[nd.inputs[j]][0] = h[nd.inputs[j]][1] = granularities[layer][0];
                        w[nd.inputs[j]][0] = w[nd.inputs[j]][1] = granularities[layer][1];
                    }
                    cal_lantency += (float)nd.cost / n_k;
                }
            } else {
                if (h[nd.outputs[0]][0] > h[nd.outputs[0]][1] && w[nd.outputs[0]][0] < w[nd.outputs[0]][1]) {
                    cal_lantency += (float)nd.cost / n_k * 2;
                } else {
                    cal_lantency += (float)nd.cost / n_k;
                }
                if (nd.is_matmul) {
                    for (int j = 0; j <= 1; j++) {
                        if (h[nd.outputs[0]][j] > h[nd.inputs[0]][0]) {
                            h[nd.inputs[0]][0] = h[nd.outputs[0]][j];
                            w[nd.inputs[0]][0] = tensors[nd.inputs[0]].width;
                            is_h[nd.inputs[0]][0] = is_h[nd.outputs[0]][j];
                        }
                        h[nd.inputs[0]][1] = h[nd.outputs[0]][j];
                        w[nd.inputs[0]][1] = tensors[nd.inputs[0]].width;
                        is_h[nd.inputs[0]][1] = is_h[nd.outputs[0]][j];


                        h[nd.inputs[1]][0] = tensors[nd.inputs[1]].height;
                        w[nd.inputs[1]][0] = w[nd.outputs[0]][j];
                        is_h[nd.inputs[1]][0] = is_h[nd.outputs[0]][j];
                        if (w[nd.outputs[0]][j] > w[nd.inputs[1]][1]) {
                            h[nd.inputs[1]][1] = tensors[nd.inputs[1]].height;
                            w[nd.inputs[1]][1] = w[nd.outputs[0]][j];
                            is_h[nd.inputs[1]][1] = is_h[nd.outputs[0]][j];
                        }
                    }
                } else {
                    for (int j = 0; j <= 1; j++) {
                        for (int k = 0; k < nd.inputs.size(); k++) {
                            if (h[nd.outputs[0]][j] > h[nd.inputs[k]][0]) {
                                h[nd.inputs[k]][0] = h[nd.outputs[0]][j];
                                w[nd.inputs[k]][0] = w[nd.outputs[0]][j];
                                is_h[nd.inputs[k]][0] = is_h[nd.outputs[0]][j];
                            }

                            if (w[nd.outputs[0]][j] > w[nd.inputs[k]][1]) {
                                h[nd.inputs[k]][1] = h[nd.outputs[0]][j];
                                w[nd.inputs[k]][1] = w[nd.outputs[0]][j];
                                is_h[nd.inputs[k]][1] = is_h[nd.outputs[0]][j];
                            }
                        }
                    }
                }
            }
        }

        float global_memory = 0.;
        float part_memory[2] = {0., 0.};
        float k_memory = 0., full_memory = 0.;
        float part_k_memory[2] = {0., 0.};
        init(retained);
        if (layer > 0) {
            for (int i = 0; i < tensors_to_retain[layer - 1].size(); i++) {
                retained[tensors_to_retain[layer - 1][i]] = 1;
            }
        }
        for (int i = 0; i < tensors_to_retain[layer].size(); i++) {
            retained[tensors_to_retain[layer][i]] = 1;
        }

        for (int i = 0; i < n; i++) {
            if (d_in[layer][i] == 0 && d_out[layer][i] > 0 && !retained[i]) {
                if (h[i][0] > h[i][1] && w[i][0] < w[i][1]) {
                    global_memory += h[i][0] * w[i][0] + h[i][1] * w[i][1];
                } else {
                    global_memory += std::max(h[i][0], h[i][1]) * std::max(w[i][0], w[i][1]);
                }
                if ((h[i][0] == tensors[i].height && w[i][0] == tensors[i].width)
                    || (h[i][1] == tensors[i].height && w[i][1] == tensors[i].width)) {
                    full_memory += h[i][0] * w[i][0];
                } else {
                    if (h[i][0] == h[i][1] && w[i][0] == w[i][1]) {
                        part_memory[is_h[i][0]] += h[i][0] * w[i][0];
                        if ((is_h[i][0] && w[i][0] == tensors[i].width) || (!is_h[i][0] && h[i][0] == tensors[i].height)) {
                            part_k_memory[is_h[i][0]] += h[i][0] * w[i][0];
                        }
                    } else {
                        if (h[i][0] > h[i][1]) {
                            part_memory[is_h[i][0]] += h[i][0] * w[i][0];
                            if ((is_h[i][0] && w[i][0] == tensors[i].width) || (!is_h[i][0] && h[i][0] == tensors[i].height)) {
                                part_k_memory[is_h[i][0]] += h[i][0] * w[i][0];
                            }
                        }
                        if (w[i][0] < w[i][1]) {
                            part_memory[is_h[i][1]] += h[i][1] * w[i][1];
                            if ((is_h[i][0] && w[i][0] == tensors[i].width) || (!is_h[i][0] && h[i][0] == tensors[i].height)) {
                                part_k_memory[is_h[i][0]] += h[i][1] * w[i][1];
                            }
                        }
                    }
                }
            }
        }

        bool output_is_retained = retained[nodes[subgraphs[layer].back()].outputs[0]];
        for (int h = 0; h < n_h; h++) {
            for (int w = 0; w < n_w; w++) {
                bool link_h, link_w;
                if (traversal_orders[layer].size() == 0) {
                    link_h = 0;
                    link_w = (w != 0);
                } else {
                    if (h == 0 && w == 0) {
                        link_h = link_w = 0;
                    } else {
                        int pre_h = traversal_orders[layer][h * n_w + w - 1] / n_w;
                        int pre_w = traversal_orders[layer][h * n_w + w - 1] % n_w;
                        int now_h = traversal_orders[layer][h * n_w + w] / n_w;
                        int now_w = traversal_orders[layer][h * n_w + w] % n_w;
                        link_h = (pre_h == now_h);
                        link_w = (pre_w == now_w);
                    }
                }
                for (int k = 0; k < n_k; k++) {
                    float turn_memory = global_memory;
                    if (!(h == 0 && w == 0 && k == 0)) {
                        turn_memory -= full_memory;
                        if (n_k > 1) {
                            if (k == 0) {
                                if (link_h) {
                                    turn_memory -= part_k_memory[0];
                                }
                                if (link_w) {
                                    turn_memory -= part_k_memory[1];
                                }
                            } else {
                                turn_memory -= part_k_memory[0];
                                turn_memory -= part_k_memory[1];
                            }
                        } else {
                            if (link_h) {
                                turn_memory -= part_memory[0];
                            }
                            if (link_w) {
                                turn_memory -= part_memory[1];
                            }
                        }
                    }
                    if (k == n_k - 1 && !output_is_retained) {
                        turn_memory += granularities[layer][0] * granularities[layer][1];
                    }
                    if (h == n_h - 1 && w == n_w - 1 && k == n_k - 1) {
                        int id = nodes[subgraphs[layer].back()].outputs[0];
                        if (total_d_out[id] == 0) {
                            if (retained[id]) {
                                turn_memory += tensors[id].height * tensors[id].width;
                            }
                        }
                    }
                    // printf("%lf %lf\n", turn_memory / slow_memory_bandwidth, cal_lantency);
                    lantency += std::max(turn_memory / slow_memory_bandwidth, cal_lantency);
                }
            }
        }
        return lantency;
    }

    void get_lantencies() {
        lantencies.resize(n_layers);
        for (int layer = 0; layer < n_layers; layer++) {
            lantencies[layer] = get_lantency(layer);
        }
    }

    float get_total_lantency() {
        float total_lantency = 0.;
        for (int layer = 0; layer < n_layers; layer++) {
            total_lantency += lantencies[layer];
        }
        return total_lantency;
    }

    std::tuple<std::vector<Tensor>, std::vector<bool>, std::vector<bool>, float> sa_solution() {
        float global_best_lantency, current_best_lantency;
        std::vector<Tensor> global_best_state;
        std::vector<bool> global_best_retain;
        std::vector<bool> global_best_granularity_orders;
        for (int i = 0; i < n; i++) {
            tensors[i].is_cut = 1;
        }
        
        retain.assign(n_layers, 0);
        granularity_orders.assign(n_layers, 0);
        get_subgraphs();
        get_retain();
        get_granularities();
        get_traversal_orders();
        get_lantencies();
        current_best_lantency = get_total_lantency();
        global_best_lantency = current_best_lantency;
        printf("%lf\n", global_best_lantency);

        global_best_retain.assign(n_layers, 0);
        global_best_granularity_orders.assign(n_layers, 0);
        global_best_state.resize(n);
        global_best_state = tensors;

        float init_temp_0 = 90000.0 * n;
        float min_temp_0 = 0.01;
        float cooling_rate_0 = 0.95;
        for (float temp_0 = init_temp_0; temp_0 >= min_temp_0;) {
            int x = ran(0, n - 1);
            tensors[x].is_cut ^= 1;
            retain.assign(n_layers, 0);
            granularity_orders.assign(n_layers, 0);
            if (!get_subgraphs()) { tensors[x].is_cut ^= 1; }
            else { get_retain(); if (!get_granularities()) { tensors[x].is_cut ^= 1;}
            else {
                temp_0 *= cooling_rate_0;

                get_traversal_orders();
                get_lantencies();
                float current_lantency = get_total_lantency();
                float best_lantency = current_lantency;
                std::vector<bool> best_retain;
                std::vector<bool> best_granularity_orders;
                best_retain.assign(n_layers, 0);
                best_granularity_orders.assign(n_layers, 0);

                float init_temp_1 = 3400.0 * n;
                float min_temp_1 = 0.01;
                float cooling_rate_1 = 0.95;
                for (float temp_1 = init_temp_1; temp_1 >= min_temp_1; temp_1 *= cooling_rate_1) {
                    if (n_layers > 1 && ran(0, 1)) {
                        int layer = ran(0, n_layers - 2);
                        retain[layer] = !retain[layer];
                        upd_retain(layer);

                        std::vector<int> temp_granularities[2];
                        float temp_lantencies[2];
                        float sub_lantency = 0.;

                        temp_granularities[0] = granularities[layer];
                        granularities[layer] = get_granularity(layer);
                        temp_granularities[1] = granularities[layer + 1];
                        granularities[layer + 1] = get_granularity(layer + 1);
                        if (granularities[layer].size() == 0 || granularities[layer + 1].size() == 0) {
                            retain[layer] = !retain[layer];
                            upd_retain(layer);
                            granularities[layer] = temp_granularities[0];
                            granularities[layer + 1] = temp_granularities[1];
                            continue;
                        }

                        upd_traversal_order(layer);
                        upd_traversal_order(layer + 1);
                        temp_lantencies[0] = get_lantency(layer);
                        sub_lantency += lantencies[layer] - temp_lantencies[0];
                        temp_lantencies[1] = get_lantency(layer + 1);
                        sub_lantency += lantencies[layer + 1] - temp_lantencies[1];
                        
                        
                        if (ran_real(0, 1) < std::exp((current_lantency - (current_lantency - sub_lantency)) / temp_1)) {
                        
                            current_lantency -= sub_lantency;

                            lantencies[layer] = temp_lantencies[0];
                            lantencies[layer + 1] = temp_lantencies[1];
                            
                            if (current_lantency < best_lantency) {
                                best_retain = retain;
                                best_granularity_orders = granularity_orders;
                                best_lantency = current_lantency;
                            }
                        } else {
                            retain[layer] = !retain[layer];
                            upd_retain(layer);
                            granularities[layer] = temp_granularities[0];
                            upd_traversal_order(layer);
                            
                            granularities[layer + 1] = temp_granularities[1];
                            upd_traversal_order(layer + 1);
                        }
                    } else {
                        int layer = ran(0, n_layers - 1);
                        granularity_orders[layer] = !granularity_orders[layer];
                        std::vector<int> temp_granularity = granularities[layer];
                        granularities[layer] = get_granularity(layer);
                        upd_traversal_order(layer);
                        float temp_lantency = get_lantency(layer);
                        float sub_lantency = lantencies[layer] - temp_lantency;
                        if (ran_real(0, 1) < std::exp((current_lantency - (current_lantency - sub_lantency)) / temp_1)) {
                            current_lantency -= sub_lantency;
                            lantencies[layer] = temp_lantency;
                            if (current_lantency < best_lantency) {
                                best_retain = retain;
                                best_granularity_orders = granularity_orders;
                                best_lantency = current_lantency;
                            }
                        } else {
                            granularity_orders[layer] = !granularity_orders[layer];
                            granularities[layer] = temp_granularity;
                            upd_traversal_order(layer);
                        }
                    }
                }
                // print(subgraphs,
                //       granularities,
                //       tensors_to_retain,
                //       traversal_orders);
                // printf("%lf\n", best_lantency);
                if (ran_real(0, 1) < std::exp((current_best_lantency - best_lantency) / temp_0)) {
                    if (best_lantency < global_best_lantency) {
                        global_best_lantency = best_lantency;
                        global_best_state = tensors;
                        global_best_retain = best_retain;
                        global_best_granularity_orders = best_granularity_orders;
                    }
                    current_best_lantency = best_lantency;
                } else {
                    tensors[x].is_cut ^= 1;
                }
            }}
        }
        printf("%lf\n", global_best_lantency);
        return {global_best_state, global_best_retain, global_best_granularity_orders, global_best_lantency};
    }

    std::vector<float> evaluate(std::vector<std::vector<int> > subgraphs,
                                std::vector<std::vector<int> > granularities,
                                std::vector<std::vector<int> > tensors_to_retain,
                                std::vector<std::vector<int> > traversal_orders) {
        this->subgraphs = subgraphs;
        get_necessity();
        this->granularities = granularities;
        this->tensors_to_retain = tensors_to_retain;
        this->traversal_orders = traversal_orders;
        get_lantencies();
        return lantencies;
    }

    std::tuple<std::vector<std::vector<int> >,
               std::vector<std::vector<int> >,
               std::vector<std::vector<int> >,
               std::vector<std::vector<int> >,
               std::vector<float> > final_solution(std::vector<Tensor>& best_state,
                                                   std::vector<bool>& best_retain,
                                                   std::vector<bool>& best_granularity_orders) {
        tensors = best_state;
        retain = best_retain;
        granularity_orders = best_granularity_orders;
        // delete
        assert(get_subgraphs());
        get_retain();
        assert(get_granularities());
        get_traversal_orders();
        get_lantencies();
        return {subgraphs, granularities, tensors_to_retain, traversal_orders, lantencies};
    }
};

void solve(const std::string file_in_name, const std::string file_out_name) {
    std::ifstream file_in(file_in_name);
    json j_in;
    file_in >> j_in; 

    std::vector<int> widths = j_in["widths"];
    std::vector<int> heights = j_in["heights"];
    std::vector<std::vector<int> > inputs = j_in["inputs"];
    std::vector<std::vector<int> > outputs = j_in["outputs"];
    std::vector<float> base_costs = j_in["base_costs"];
    std::vector<std::string> op_types = j_in["op_types"];
    float fast_memory_capacity = j_in["fast_memory_capacity"];
    float slow_memory_bandwidth = j_in["slow_memory_bandwidth"];
    std::vector<int> native_granularity = j_in["native_granularity"];

    // auto start = std::chrono::high_resolution_clock::now();
    ScheduleEvaluator myEvaluater(
        widths,
        heights,
        inputs,
        outputs,
        base_costs,
        op_types,
        fast_memory_capacity,
        slow_memory_bandwidth,
        native_granularity
    );

    std::vector<Tensor> best_state;
    std::vector<bool> best_retain;
    std::vector<bool> best_granularity_orders;
    float best_lantency = INF_LANTENCY;

    for (int t = 0; t < 30; t++) {
        auto [current_state, current_retain, current_granularity_orders, current_lantency] = 
                                    myEvaluater.sa_solution();
        if (current_lantency < best_lantency) {
            best_state = current_state;
            best_retain = current_retain;
            best_granularity_orders = current_granularity_orders;
            best_lantency = current_lantency;
        }
    }

    printf("best lantency: %lf\n", best_lantency);

    auto [subgraphs, granularities, tensors_to_retain, traversal_orders, lantencies] =
            myEvaluater.final_solution(best_state, best_retain, best_granularity_orders);
    // print(subgraphs, granularities, tensors_to_retain, traversal_orders);

    // auto end = std::chrono::high_resolution_clock::now();
    // auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);

    // std::cout << "time:" << duration.count() << " ms" << std::endl;

    nlohmann::ordered_json j_out;
    j_out["subgraphs"] = subgraphs;
    j_out["granularities"] = granularities;
    j_out["tensors_to_retain"] = tensors_to_retain;
    for (int i = 0; i < traversal_orders.size(); i++) {
        j_out["traversal_orders"].push_back(
            traversal_orders[i].empty() ? nlohmann::ordered_json(nullptr) : nlohmann::ordered_json(traversal_orders[i])
        );
    }
    j_out["subgraph_latencies"] = lantencies;
    std::ofstream file_out(file_out_name);
    file_out << j_out.dump(4);
    file_out.close();
}
